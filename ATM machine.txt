#ATM machine-minimum currency notes
a=int(input("Enter the amount you want to withdraw :"))
b=a//2000#notes of 2000
c=a%2000
d=c//500#notes of 500
e=c%500
f=e//200#notes of 200
g=e%200  
h=g//100#notes of 100
i=g%100
j=i//50#notes of 50
k=i%50
l=k//20#notes of 20
m=k%20
n=m//10#notes of 10
o=m%10
p=o//5#notes of 5
q=o%5
r=q//2#notes of 2
s=q%2#notes of 1
print(f"2000: {b}\n500: {d}\n200: {f}\n100: {h}\n50: {j}\n20: {l}\n10: {n}\n5: {p}\n2: {r}\n1: {s}")
